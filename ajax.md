#include "TrafficLightStates.h"

// Instantiate all state objects
RedState   redState;
YellowState yellowState;
GreenState greenState;
BlinkState blinkState;

// --- Red State ---
void RedState::enter() {
  digitalWrite(RED_PIN, HIGH);
  digitalWrite(YELLOW_PIN, LOW);
  digitalWrite(GREEN_PIN, LOW);
  startTime = millis();
}

void RedState::update() {}
TrafficLightState* RedState::next() {
  if (elapsed(RED_DURATION)) return &yellowState;
  return this;
}

// --- Yellow State ---
bool YellowState::fromRed = true;

void YellowState::enter() {
  digitalWrite(RED_PIN, LOW);
  digitalWrite(YELLOW_PIN, HIGH);
  digitalWrite(GREEN_PIN, LOW);
  startTime = millis();
}

void YellowState::update() {}
TrafficLightState* YellowState::next() {
  if (elapsed(YELLOW_DURATION)) {
    TrafficLightState* next = fromRed ? (TrafficLightState*)&greenState : (TrafficLightState*)&redState;
    fromRed = !fromRed;
    return next;
  }
  return this;
}

// --- Green State ---
void GreenState::enter() {
  digitalWrite(RED_PIN, LOW);
  digitalWrite(YELLOW_PIN, LOW);
  digitalWrite(GREEN_PIN, HIGH);
  startTime = millis();
}

void GreenState::update() {}
TrafficLightState* GreenState::next() {
  if (elapsed(GREEN_DURATION)) return &blinkState;
  return this;
}

// --- Blink State ---
void BlinkState::enter() {
  digitalWrite(RED_PIN, LOW);
  digitalWrite(YELLOW_PIN, LOW);
  digitalWrite(GREEN_PIN, HIGH);
  startTime = millis();
  lastToggle = millis();
  ledOn = true;
}

void BlinkState::update() {
  if (millis() - lastToggle >= GREEN_BLINK_INTERVAL) {
    ledOn = !ledOn;
    digitalWrite(GREEN_PIN, ledOn ? HIGH : LOW);
    lastToggle = millis();
  }
}

TrafficLightState* BlinkState::next() {
  if (elapsed(GREEN_BLINK_DURATION)) {
    digitalWrite(GREEN_PIN, LOW);
    return &yellowState;
  }
  return this;
}
